class Solution {
public int longestContinuousSubstring(String s) {
int cnt=1,max=1,n=s.length();
for(int i=1;i <n;i++)
{
if(s.charAt(i)==(s.charAt(i-1)+1))
cnt++;
else
{
cnt=1;
}
if(cnt>max)
max=cnt;
}
return max;
}
}
